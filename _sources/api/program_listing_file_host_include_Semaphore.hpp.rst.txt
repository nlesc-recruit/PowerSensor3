   namespace PowerSensor3 {
   
   class Semaphore {
    public:
       explicit Semaphore(unsigned initialLevel = 0): level(initialLevel) {}
   
       void up(unsigned count = 1) {
         std::unique_lock<std::mutex> lock(mutex);
         level += count;
         if (count == 1)
           cv.notify_one();
         else
           cv.notify_all();
       }
   
       void down(unsigned count = 1) {

         level -= count;
       }

       std::condition_variable cv;

   };
